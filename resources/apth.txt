#!/bin/bash
# apth - Application Power Tool HermiTech 4.2b
# by goodgamemaga | Russian build (encoding fixed)
# Installs apt packages without root into $HOME/linux using dpkg -x

HOMEA=$HOME/linux

echo 'APTH - Application Power Tool HermiTech 4.2b (goodgamemaga)'
echo 'This tool installs apps without root'

if ! [ -d $HOME/linux ]; then
    echo 'Creating linux directory...'
    mkdir -p $HOME/linux
fi

# Fail on any error
set -e

BUILD_DIR=$HOMEA
CACHE_DIR=$HOMEA/tmp

APT_CACHE_DIR="$CACHE_DIR/apt/cache"
APT_STATE_DIR="$CACHE_DIR/apt/state"
APT_SOURCELIST_DIR="$CACHE_DIR/apt/sources"
APT_SOURCES="$APT_SOURCELIST_DIR/sources.list"

APT_OPTIONS="-o debug::nolocking=true \
             -o dir::cache=$APT_CACHE_DIR \
             -o dir::state=$APT_STATE_DIR \
             -o dir::etc::sourcelist=$APT_SOURCES"

rm -rf $APT_CACHE_DIR
mkdir -p "$APT_CACHE_DIR/archives/partial"
mkdir -p "$APT_STATE_DIR/lists/partial"
mkdir -p "$APT_SOURCELIST_DIR"
cat /etc/apt/sources.list > "$APT_SOURCES"

echo "-----> Updating apt caches"
apt-get $APT_OPTIONS update 2>&1 | sed 's/^/       /'

for PACKAGE in $*; do
    if [[ $PACKAGE == *.deb ]]; then
        PACKAGE_NAME=$(basename $PACKAGE .deb)
        PACKAGE_FILE=$APT_CACHE_DIR/archives/$PACKAGE_NAME.deb
        echo "-----> Fetching $PACKAGE"
        curl -s -L -z "$PACKAGE_FILE" -o "$PACKAGE_FILE" "$PACKAGE" 2>&1 | sed 's/^/       /'
    else
        echo "-----> Fetching .debs for $PACKAGE"
        apt-get $APT_OPTIONS -y --force-yes -d install --reinstall $PACKAGE 2>&1 | sed 's/^/       /'
    fi
done

mkdir -p $BUILD_DIR

for DEB in $(ls -1 $APT_CACHE_DIR/archives/*.deb 2>/dev/null); do
    echo "-----> Installing $(basename $DEB)"
    dpkg -x $DEB $BUILD_DIR
done

echo "-----> Installation complete"

# Clean up cache
rm -rf $CACHE_DIR/*

# Fix pkg-config prefix paths
find $BUILD_DIR -type f -ipath '*/pkgconfig/*.pc' \
    | xargs --no-run-if-empty -n 1 sed -i \
      -e 's!^prefix=\(.*\)$!prefix='"$BUILD_DIR"'/.\1!g'

# Export environment so installed binaries are usable immediately
echo "-----> Setting PATH and library exports"
LINE1=$(find $HOME/linux -type d | awk '{printf "%s:", $0}')
export LD_LIBRARY_PATH=$LINE1
export LIBRARY_PATH=$LINE1
export PATH=$LINE1:$PATH
